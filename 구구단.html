package exam8;

public class GuGuDan2 {
	  public static void main(String[] args) {
		    for (int i = 2; i <= 9; i++) {
		      for (int j = 1; j <= 9; j++) {
		        System.out.print(i + " * " + j + " = " + String.format("%2d", i * j));
		        System.out.print("    ");
		      }
		      System.out.println();
		    }
		  }
		}
